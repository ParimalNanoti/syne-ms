package com.example.cart_api_gateway;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.cloud.client.discovery.EnableDiscoveryClient;
import org.springframework.cloud.gateway.route.RouteLocator;
import org.springframework.cloud.gateway.route.builder.RouteLocatorBuilder;
import org.springframework.context.annotation.Bean;

@SpringBootApplication
@EnableDiscoveryClient
public class CartApiGatewayApplication {

	public static void main(String[] args) {
		SpringApplication.run(CartApiGatewayApplication.class, args);
	}
	@Bean
	RouteLocator gatewayRoute(RouteLocatorBuilder builder) {
		return builder.routes()
				.route("sr",rs-> rs
						.path("/shoppingcart/**")
						.uri("lb://shop-cart-app"))
						.build();}

}
